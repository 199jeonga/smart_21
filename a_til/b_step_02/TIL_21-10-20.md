##### 하나의 행동 처리를 위한 틀 잡기 

![](img/img_22.png)

```js
// 틀 제작
var yearSelector = document.querySelector('#year');
var y = 2021;
var creatOption;
var makeFn = function(){
    createOption = document.createElement('option');
    createOption.innerText = y + '년';
    createOption.value = y;
    yearSelector.append(createOption)
};

// 틀 수행
makeFn()
y = y -1;
makeFn()
y = y -1;
```
